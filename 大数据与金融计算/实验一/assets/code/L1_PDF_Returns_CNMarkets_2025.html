def myfun_emp_pdf_linx(sample, num_bin):
    # 计算样本的概率密度, 计算等距区间的边界
    bin = np.linspace(np.min(sample), np.max(sample), num_bin)
    # 计算每个区间的概率密度
    x_pdf = np.zeros(num_bin-1)
    y_pdf = np.zeros(num_bin-1)
    for i in range(num_bin-1):
        x_pdf[i] = (bin[i]+bin[i+1])/2
        y_pdf[i] = np.sum((sample>=bin[i])&(sample<bin[i+1]))/len(sample)/(bin[i+1]- bin[i])
    return x_pdf, y_pdf

def myfun_fit_norm(sample):
    # 拟合正态分布
    mu, sigma = norm.fit(sample)
    x_norm = np.linspace(np.min(sample), np.max(sample), 100)
    y_norm = norm.pdf(x_norm, mu, sigma)
    return x_norm, y_norm
